# src/components/Todo.vue

//computed
computed: {
    activeTodoId: {
        get: function() {
            return this.$store.state.activeTodoId;
        },
    },
    record: {
        get: function() {
            return this.$store.state.record;
        },
        set: function(value){
            this.setRecord(value)
            // this.$store.commit('setRecord',value)
        }
    }
},

//method
methods: {
    // 当用户点击回车时候 ，给items的值新增一个对象，this.text 即输入框绑定的值
    onAdd() {
        this.items.push({
            checked: false,
            text: this.text,
            isDelete: false
        });
        //初始化输入框的值。
        this.text = "";
    },
    init() {
        this.todo = this.$store.state.todoList.find(i => i.id == this.activeTodoId)
        this.record = this.todo.record
    },
    ...mapMutations([
        'setRecord'
    ])
},

//created
created(){
    this.todo = this.$store.state.todoList.find(i => i.id == this.todoId)
    this.items = this.todo.record
}

// watch
watch: {
    activeTodoId: function() {
        this.init()
    }
}

# todo删除

# src/components/Todo.vue

<a class=" nav-item">
    <!-- 删除图标-->
    <span class="icon-trash" @click="deleteTodo"></span>
</a>

deleteTodo(){
    for (let i in this.todoList) {
        if(this.todoList[i].id === this.activeTodoId){
            this.todoList.splice(i,1)
        }
    }
    if(this.todoList.length > 0){
        this.activeTodoId = this.todoList[0].id
        this.init
    }else{
        this.todo = {
            title: '未命名',
            count: 0
        }
        this.addTodo(this.todo)
        this.record = []
    }
},

deleteRecord(item){
    // console.log(item)

        if(this.record[i].id === item.id){
            this.record.splice(i,1)
        }

editRecord(item) {
    // console.log(item)
    for (let i in this.record) {
        if(this.record[i].id === item.id){
            this.record[i].text = item.text
        }
    }
